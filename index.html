  function makeCard(m){

    const color = tagColors[Math.floor(Math.random()*tagColors.length)];

    const card = document.createElement('article');

    card.className = 'postcard';

    card.innerHTML = `

      <h4>${m.title}</h4>

      <p>${m.story}</p>

      <div class="author">— ${m.name}</div>

    `;

    return card;

  }


  seedMemories.forEach(m => grid.appendChild(makeCard(m)));
